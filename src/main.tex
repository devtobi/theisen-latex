	\subsection{Aufzählungen}\label{subsec:lists}
	
	Aufzählungen ohne voranstehende Beschriftung
	
	\begin{description}
		\item{Item 1}
		\item{Item 2}
		\item{Item 3}
	\end{description}
	
	Ungeordnete Aufzählungen
	
	\begin{itemize}
		\item{Item 1}
		\item{Item 2}
		\item{Item 3}
	\end{itemize}
	
	Geordnete Aufzählungen 
	
	\begin{enumerate} %[a/1/I/A] optional

			\item Auch verschachtelt möglich

		\end{enumerate}

		\item{Item 3}
	\end{enumerate}
